\begin{aligned}
&\text{maximize }\mathbf{c}^{T}\mathbf{x}\\
&\text{subject to }\mathbf{A}\mathbf{x} \leq \mathbf{b}, \mathbf{x} \geq 0
\end{aligned}
\Longleftrightarrow
\begin{aligned}
&\text{minimize }\mathbf{y}^{T}\mathbf{b}\\
&\text{subject to }\mathbf{y}^{T}\mathbf{A} \geq \mathbf{c}^{T}, \mathbf{y} \geq 0
\end{aligned}
\end{equation*}
