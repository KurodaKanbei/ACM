	}

	}
};
Point crs(const Line & a, const Line & b) {
	double lambda((b.s - a.s) * b.d / (a.d * b.d));
	return a.s + lambda * a.d;

		double len1(c[prv].d().len()), len2(d().len()), len3(c[nxt].d().len());
